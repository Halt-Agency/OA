function cleanup_old_candidates() {
    $jwt = get_jwt_token();
    
    // Search for old candidates in TrackerRMS
    $search_body = [
        'updatedBefore' => date('c', strtotime('-7 months')), // 7 months = 6 + 1 buffer
        'maxResults' => 100
    ];
    
    $candidates = search_resources( $search_body, $jwt );
    
    foreach ( $candidates as $candidate ) {
        // Check if they've been warned
        $warned = get_option( 'warned_candidate_' . $candidate['id'] );
        
        if ( ! $warned ) {
            // Send warning email: "We'll delete your data in 30 days"
            send_deletion_warning_email( $candidate['contactDetails']['email'], $candidate['id'] );
            update_option( 'warned_candidate_' . $candidate['id'], time() );
        } else {
            // Check if 30 days passed since warning
            $warning_date = $warned;
            if ( time() - $warning_date > 30 * DAY_IN_SECONDS ) {
                // DELETE from TrackerRMS
                $url = 'https://evoglapi.tracker-rms.com/api/v1/Resource/' . $candidate['id'];
                
                wp_remote_request( $url, [
                    'method' => 'DELETE',
                    'headers' => [
                        'Authorization' => 'Bearer ' . $jwt
                    ]
                ]);
                
                // Log deletion
                log_gdpr_deletion( 'Resource', $candidate['id'], 'Retention period expired' );
                delete_option( 'warned_candidate_' . $candidate['id'] );
            }
        }
    }
}

function process_erasure_request( $email ) {
    $jwt = get_jwt_token();
    
    // Search TrackerRMS by email
    $search_body = [ 'email' => $email, 'maxResults' => 10 ];
    
    // Check all three record types
    $contacts = search_contacts( $search_body, $jwt );
    $resources = search_resources( $search_body, $jwt );
    $leads = search_leads( $search_body, $jwt );
    
    // Delete from TrackerRMS
    foreach ( $contacts as $contact ) {
        delete_tracker_record( 'Contact', $contact['contactId'], $jwt );
        log_gdpr_deletion( 'Contact', $contact['contactId'], 'User requested erasure' );
    }
    
    foreach ( $resources as $resource ) {
        delete_tracker_record( 'Resource', $resource['resourceId'], $jwt );
        log_gdpr_deletion( 'Resource', $resource['resourceId'], 'User requested erasure' );
    }
    
    foreach ( $leads as $lead ) {
        delete_tracker_record( 'Lead', $lead['leadId'], $jwt );
        log_gdpr_deletion( 'Lead', $lead['leadId'], 'User requested erasure' );
    }
    
    // Delete from WordPress
    delete_wordpress_attendee_records( $email );
    delete_wordpress_form_submissions( $email );
    
    // Send confirmation
    send_erasure_confirmation_email( $email );
    
    // Log for audit trail
    log_gdpr_action( 'ERASURE', $email, 'All records deleted from WordPress and TrackerRMS' );
}

function delete_tracker_record( $type, $id, $jwt ) {
    $url = "https://evoglapi.tracker-rms.com/api/v1/{$type}/{$id}";
    
    $response = wp_remote_request( $url, [
        'method' => 'DELETE',
        'headers' => [
            'Authorization' => 'Bearer ' . $jwt
        ],
        'timeout' => 20
    ]);
    
    $code = wp_remote_retrieve_response_code( $response );
    return ( 200 === $code || 204 === $code ); // Both mean deleted/doesn't exist
}
